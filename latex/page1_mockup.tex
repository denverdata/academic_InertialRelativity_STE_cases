\subsection*{The Framework in Brief}

\begin{align*}
D        &= M/R                      &&\text{(inertial density)}\\
D_{crit} &= c^2/2G = \num{6.73295e26}\,\unit{\kilogram\per\meter} &&\text{(threshold)}\\
k        &= D/D_{crit} = r_{s}/R = v^2/c^2 &&\text{(scale factor)}\\
GTD      &= \sqrt{1-k}               &&\text{(gravitational TD)}\\
DTD      &= \sqrt{k}                 &&\text{(DeGerlia TD)}\\
DTD^2 &+ GTD^2 = 1                   &&\text{(unit relation)}
\end{align*}
